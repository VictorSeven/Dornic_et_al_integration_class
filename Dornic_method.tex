We consider the stochastic equation

\begin{equation}\label{eq:fisher_stoch2}
\dot{\rho}(x,t)= D\nabla^2 \rho + a
\rho-bf^2+\gamma\sqrt{\rho}\xi(x,t)
\end{equation}

where $\rho(x,t)$ is the population density at space $x$ and time $t$, and
$\xi(x,t)$ a Gaussian white noise satisfying $\langle
\xi(x,t)\rangle=0$ and  $\langle
\xi(x,t) \xi(x',t')\rangle=\delta(x-x')\delta(t-t')$.\\

We perform a two-step numerical integration (see \cite{dornic2005integration,weissmann2018simulation}):

\begin{enumerate}
 \item {\it{Non-linear and diffusion terms}}.
 Integration of $\dot{\rho}(x,t)= D\nabla^2 \rho - b \rho^2$ is done
 by employing a finite-difference fourth-order Runge-Kutta method
 obtaining a first solution $\rho^*$.
 
 \item {\it{Linear and stochastic terms}}. The term
 $a \rho+\gamma\sqrt{\rho}\xi(x,t)$ is integrated
 in an exact way as \cite{dornic2005integration}:
 
 \begin{equation}
  \rho(x,t)=r_\mathrm{Gamma}\{r_\mathrm{Poisson}\{\lambda \rho^*(x,t) e^{at}\}\}/\lambda.
 \end{equation}


 being $\lambda=\frac{2a}{\gamma^2(e^{at}-1)}$, and
 $r_\mathrm{Gamma}$, $r_\mathrm{Poisson}$ random values 
 obtained from the Gamma and Poisson probability distributions, i.e. 
 $Prob[r_\mathrm{Gamma}(a)=z]=\frac{z^{a-1}e^{-z}}{\Gamma [a]}$
 and $Prob[r_\mathrm{Poisson}(a)=z]=\frac{a^ze^{-a}}{z!}$, respectively. 
\end{enumerate}
